    def getValueByType()
      case @type
      when :line ;
        return getValueByType_Line() ;
      when :box ;
        return getValueByType_Box() ;
      when :ellipse ;
        return getValueByType_Ellipse() ;
      when :text ;
        return getValueByType_Text() ;
      when :image ;
        return getValueByType_Image() ;
      end
    end
