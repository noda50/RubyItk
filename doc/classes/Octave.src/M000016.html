  def evalCodeOnStream(code)
    delimiter = DefaultDelimiter ;
    ans = "" ;
    fullcode = code + ";disp(\"#{delimiter}\")" ;
#    p [:puts, fullcode]
    @strm.puts(fullcode) ;
    while(line = @strm.gets)
#      p [:gets, line]
      break if(line =~ /^#{delimiter}$/) ;
      ans += line ;
    end
    ans
  end
